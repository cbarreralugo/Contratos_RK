ALTER PROCEDURE [dbo].[sp_ctr_obtener_custodio]
AS
BEGIN
    SELECT c.id_tipo_custodio as id,
	c.nombre_custodio as 'Nombre custodio',
	p.nombre_valor as 'Tipo', 
	c.activo as activo,
	tc.nombre_contrato as 'Tipo contrato'
	FROM tb_ctr_tipo_custodio_rk_c c
	inner join tb_ctr_tipo_valor_rk_c p on c.id_tipo_valor=p.id_tipo_valot
	inner join tb_ctr_tipo_contrato_rk_c tc on c.id_contrato= tc.id_tipo_contrato
END

ALTER PROCEDURE [dbo].[sp_ctr_custodio_detalle]
    @id_tipo_custodio INT
AS
BEGIN
    SET NOCOUNT ON; 
   SELECT p.nombre_plantilla as 'valor', 
		case 
			when p.id_tipo_valor =1 then 'BANCOSANTA'
			else 'SAMMX'
		end
		as 'codigo'
    FROM tb_ctr_tipo_custodio_rk_c c 
	inner join tb_ctr_valor_plantilla_rk_c p on c.id_tipo_custodio = p.id_valor_plantilla
    WHERE c.id_tipo_custodio = @id_tipo_custodio and c.activo=1
END
